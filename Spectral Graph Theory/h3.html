# creating the function to return the degree of vertices of a graph.
def degree_matrix(graph):
    v= len(graph.vs)
    degree_matrix = np.zeros(shape=(v,v))
    # degree_matrix = [[0 for _ in range(v)] for _ in range(v)]
    # degree_matrix[i][i] = [graph.degree(i) for i in range(v)]
    for i in range(v):
        degree_matrix[i][i]= graph.degree(i)
    return degree_matrix

  y = np.sort(np.linalg.eigvals(graph.data))
  print(y)
  plt.plot(y)

def eigenvalue_lap(lap_data):
  # y = np.sort(np.linalg.eigvals(lap_data))
  y = np.around(np.sort(np.linalg.eigvals(lap_data)), decimals = 3)
  print(y)
  plt.plot(y)
  plt.ylabel("Laplacian Eigenvalue")
  plt.show()
